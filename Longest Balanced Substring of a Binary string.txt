class Solution:
    def findTheLongestBalancedSubstring(self, s: str) -> int:
        answer = 0
        pos = 0

        while pos < len(s):
            start_zeros = 0

            while pos < len(s) and s[pos] == '0':
                start_zeros += 1
                pos += 1

            following_ones = 0

            while pos < len(s) and s[pos] == '1':
                following_ones += 1
                pos += 1

            balanced_length = 2 * min(start_zeros, following_ones)
            answer = max(answer, balanced_length)

        return answer
